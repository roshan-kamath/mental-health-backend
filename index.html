#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX 100
#define HASH_SIZE 10

// ---------------- STACK (Recent moods) ----------------
typedef struct {
    char moods[MAX][50];
    int top;
} Stack;

void initStack(Stack *s) {
    s->top = -1;
}

void push(Stack *s, char mood[]) {
    if (s->top == MAX - 1) {
        printf("Stack full! Cannot add mood.\\n");
        return;
    }
    strcpy(s->moods[++s->top], mood);
}

void displayStack(Stack *s) {
    printf("\\nRecent Mood Changes (Stack - LIFO):\\n");
    for (int i = s->top; i >= 0; i--) {
        printf("%s\\n", s->moods[i]);
    }
}

// ---------------- QUEUE (Long-term history) ----------------
typedef struct {
    char moods[MAX][50];
    int front, rear;
} Queue;

void initQueue(Queue *q) {
    q->front = q->rear = -1;
}

void enqueue(Queue *q, char mood[]) {
    if (q->rear == MAX - 1) {
        printf("Queue full! Cannot add mood.\\n");
        return;
    }
    if (q->front == -1) q->front = 0;
    strcpy(q->moods[++q->rear], mood);
}

void displayQueue(Queue *q) {
    printf("\\nLong-Term Mood History (Queue - FIFO):\\n");
    for (int i = q->front; i <= q->rear; i++) {
        printf("%s\\n", q->moods[i]);
    }
}

// ---------------- HASH MAP (Triggers & Symptoms) ----------------
typedef struct node {
    char trigger[50];
    char symptom[50];
    struct node* next;
} Node;

Node* hashTable[HASH_SIZE];

int hashFunction(char key[]) {
    int sum = 0;
    for (int i = 0; key[i] != '\\0'; i++)
        sum += key[i];
    return sum % HASH_SIZE;
}

void insertHash(char trigger[], char symptom[]) {
    int idx = hashFunction(trigger);
    Node* newNode = (Node*) malloc(sizeof(Node));
    strcpy(newNode->trigger, trigger);
    strcpy(newNode->symptom, symptom);
    newNode->next = hashTable[idx];
    hashTable[idx] = newNode;
}

void displayHash() {
    printf("\\nTriggers and Symptoms (Hash Map):\\n");
    for (int i = 0; i < HASH_SIZE; i++) {
        Node* temp = hashTable[i];
        if (temp != NULL) {
            printf("[%d]:\\n", i);
            while (temp != NULL) {
                printf("Trigger: %s --> Symptom: %s\\n", temp->trigger, temp->symptom);
                temp = temp->next;
            }
        }
    }
}

// ---------------- MAIN ----------------
int main() {
    Stack recent;
    Queue history;
    initStack(&recent);
    initQueue(&history);

    int choice;
    char mood[50], trigger[50], symptom[50];

    while (1) {
        printf("\\n--- AI Mental Health Mood Tracker ---\\n");
        printf("1. Add today's mood\\n");
        printf("2. View recent moods (Stack)\\n");
        printf("3. View long-term mood history (Queue)\\n");
        printf("4. Add trigger & symptom (Hash Map)\\n");
        printf("5. View triggers & symptoms\\n");
        printf("6. Exit\\n");
        printf("Enter choice: ");
        scanf("%d", &choice);
        getchar();

        switch (choice) {
            case 1:
                printf("Enter your mood: ");
                fgets(mood, 50, stdin);
                mood[strcspn(mood, "\\n")] = '\\0';
                push(&recent, mood);
                enqueue(&history, mood);
                break;

            case 2:
                displayStack(&recent);
                break;

            case 3:
                displayQueue(&history);
                break;

            case 4:
                printf("Enter trigger: ");
                fgets(trigger, 50, stdin);
                trigger[strcspn(trigger, "\\n")] = '\\0';

                printf("Enter symptom: ");
                fgets(symptom, 50, stdin);
                symptom[strcspn(symptom, "\\n")] = '\\0';

                insertHash(trigger, symptom);
                break;

            case 5:
                displayHash();
                break;

            case 6:
                printf("Thank you!\\n");
                return 0;

            default:
                printf("Invalid choice!\\n");
        }
    }
}
